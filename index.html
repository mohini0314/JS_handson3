Note: You have to complete Find_Five. No need to take any input. 

    const Find_Five = (N) =>
    { 
      let count = 0;
      let str = N+ " ";
      let letter = "5";
      for (let i = 0; i< str.length; i++)
      {
        if (str.charAt(i) == letter) 
        {
          count += 1;
        }
      }
      return count;
    };
